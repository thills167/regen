308  inline void rotate(GLfloat angle, GLfloat x_, GLfloat y_, GLfloat z_)
309  {
310  GLfloat c = cos(angle);
311  GLfloat s = sin(angle);
312  Vec3<GLfloat> rotated(
313  (x_*x_*(1-c) + c) * x
314  + (x_*y_*(1-c) - z_*s) * y
315  + (x_*z_*(1-c) + y_*s) * z,
316  (y_*x_*(1-c) + z_*s) * x
317  + (y_*y_*(1-c) + c) * y
318  + (y_*z_*(1-c) - x_*s) * z,
319  (x_*z_*(1-c) - y_*s) * x
320  + (y_*z_*(1-c) + x_*s) * y
321  + (z_*z_*(1-c) + c) * z
322  );
323  x = rotated.x;
324  y = rotated.y;
325  z = rotated.z;
326  }
