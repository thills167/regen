303  inline void rotate(GLfloat angle, GLfloat x_, GLfloat y_, GLfloat z_)
304  {
305  GLfloat c = cos(angle);
306  GLfloat s = sin(angle);
307  Vec3<GLfloat> rotated(
308  (x_*x_*(1-c) + c) * x
309  + (x_*y_*(1-c) - z_*s) * y

311  (y_*x_*(1-c) + z_*s) * x
312  + (y_*y_*(1-c) + c) * y
313  + (y_*z_*(1-c) - x_*s) * z,
314  (x_*z_*(1-c) - y_*s) * x
315  + (y_*z_*(1-c) + x_*s) * y
316  + (z_*z_*(1-c) + c) * z
317  );
318  x = rotated.x;
