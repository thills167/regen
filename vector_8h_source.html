307  inline void rotate(GLfloat angle, GLfloat x_, GLfloat y_, GLfloat z_)
308  {
309  GLfloat c = cos(angle);
310  GLfloat s = sin(angle);
311  Vec3<GLfloat> rotated(
312  (x_*x_*(1-c) + c) * x
313  + (x_*y_*(1-c) - z_*s) * y
314  + (x_*z_*(1-c) + y_*s) * z,
315  (y_*x_*(1-c) + z_*s) * x
316  + (y_*y_*(1-c) + c) * y
317  + (y_*z_*(1-c) - x_*s) * z,
318  (x_*z_*(1-c) - y_*s) * x
319  + (y_*z_*(1-c) + x_*s) * y
320  + (z_*z_*(1-c) + c) * z
321  );
322  x = rotated.x;
323  y = rotated.y;
324  z = rotated.z;
