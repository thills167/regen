304  inline void rotate(GLfloat angle, GLfloat x_, GLfloat y_, GLfloat z_)
305  {
306  GLfloat c = cos(angle);
307  GLfloat s = sin(angle);
308  Vec3<GLfloat> rotated(
309  (x_*x_*(1-c) + c) * x
310  + (x_*y_*(1-c) - z_*s) * y
311  + (x_*z_*(1-c) + y_*s) * z,
312  (y_*x_*(1-c) + z_*s) * x
313  + (y_*y_*(1-c) + c) * y
314  + (y_*z_*(1-c) - x_*s) * z,
315  (x_*z_*(1-c) - y_*s) * x
316  + (y_*z_*(1-c) + x_*s) * y
317  + (z_*z_*(1-c) + c) * z
318  );
319  x = rotated.x;

321  z = rotated.z;
322  }
