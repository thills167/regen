61 template<typename StackType, typename ValueType, typename ApplyFunc> class StateStack
62 {
63 public:
68  StateStack(ApplyFunc apply, ApplyFunc lockedApply)
69  : apply_(apply),
70  applyPtr_(apply),
71  lockedApplyPtr_(lockedApply),
72  lockCounter_(0),
73  lastNode_(NULL)
74  {

76  doApply_ = &applyInit;
77  }

79  {

89  const ValueType& value()
90  {
91  if(!stack_.isEmpty()) {

104  void push(const ValueType &v)
105  {
106  doApply_(self_,v);
107  doApply_ = &applyFilled;
108 
109  stack_.push(v);
110  }

115  void pop()
116  {
117  typename Stack<ValueType>::Node *top = stack_.topNode();
118  typename Stack<ValueType>::Node *next = top->next_;
119  if(next) {
120  if(stack_.top() != next->value_) { self_->apply(next->value_); }
121  stack_.pop();
122  }
123  else { // last value. keep the node until next push
124  stack_.popKeepNode();
125  lastNode_ = top;
126  doApply_ = &applyEmpty;
127  }
128  }

133  void lock()
134  {
135  ++lockCounter_;
136  apply_ = lockedApplyPtr_;
137  }

142  void unlock()
143  {
144  --lockCounter_;
145  if(lockCounter_<1) {
146  apply_ = applyPtr_;
147  }
148  }

153  GLboolean isLocked() const
154  {
155  return lockCounter_>0;
156  }

158 protected:
159  // cast to parent class
160  StackType *self_;
161  // just in case nothing was pushed
162  ValueType zeroValue_;
163  // The actual value stack.
164  Stack<ValueType> stack_;
165  // Function to apply the value.
166  ApplyFunc apply_;
167  // Points to actual apply function when the stack is locked.
168  ApplyFunc applyPtr_;
169  // Points to locked apply function.

375  void push(const ValueType &v_)
376  {
377  StampedValue<ValueType> v(v_,counter_.x);
378 
379  if(counter_.y>0) {
380  // an indexed value was pushed before
381  // if the indexed value was pushed before the last global push. only
382  // apply when the value of last global push is not equal
383  if(lastStamp_.top()>lastStampi_.top())
384  { doApply_(this,v_); }
385  // else an indexed value was pushed. always call apply even if some
386  // indices may already contain the right value
387  else
388  { apply_(v_); }
389  }
390  else {
391  // no indexed push was done before
392  doApply_(this,v_);
393  }
394  doApply_ = &applyFilledStamped;
395 
396  stack_.push(v);
397  lastStamp_.push(v.stamp);
398 
399  // count number of pushes for value stamps
400  counter_.x += 1;
401  }
407  void push(GLuint index, const ValueType &v_)
408  {
409  StampedValue<ValueType> v(v_,counter_.y);
410  IndexedStack &stacki = stackIndex_[index];
411 
412  if(counter_.x>counter_.y) {
413  GLint lastStampi = (stacki.isEmpty() ? -1 : stacki.top().stamp);
414  // an global value was pushed before
415  // if the global value was pushed before the last indexed push. only
416  // apply when the value of last indexed push is not equal
417  if(lastStampi>lastStamp_.top())
418  { doApplyi_[index](this,index,v_); }
419  // else an global value was pushed. call apply if values not equal
420  else if(v_ != globalValue())
421  { applyi_(index,v_); }
422  }
423  else {
424  // no global push was done before
425  doApplyi_[index](this,index,v_);
426  }
427  doApplyi_[index] = &applyFilledStampedi;
428 
429  stacki.push(v);
430  lastStampi_.push(v.stamp);
431 
432  // count number of pushes for value stamps
433  counter_.x += 1;
434  counter_.y += 1;
435  }

440  void pop()
441  {
442  IndexedStackNode *top = stack_.topNode();
443  IndexedStackNode *next = top->next_;
444  ValueType *globalValue;
445 
446  // apply previously pushed global value
447  if(next) {
448  globalValue = &next->value_.v;
449 
450  if(stack_.top().v != (*globalValue)) { apply_(*globalValue); }
451  stack_.pop();
452  }
453  else { // last value. keep the node until next push
454  globalValue = &top->value_.v;
455 
456  stack_.popKeepNode();
457  lastNode_ = top;
458  doApply_ = &applyEmptyStamped;
459  }
460  lastStamp_.pop();
461 
462  // if there are indexed values pushed we may
463  // have to re-enable them here...
464  if(counter_.y>0) {
465  // Indexed states only applied with stamp>lastEqStamp
466  // Loop over all indexed stacks and compare stamps of top element.
467  for(register GLuint i=0; i<numIndices_; ++i)
468  {
469  IndexedStack &stacki = stackIndex_[i];
470  if(stacki.isEmpty()) { continue; }
471 
472  const StampedValue<ValueType>& top = stacki.top();
473  if((GLint)top.stamp>lastStamp_.top() && (*globalValue)!=top.v)
474  { applyi_(i,top.v); }
475  }
476  }
477 
478  // count number of pushes for value stamps
479  counter_.x -= 1;
480  }
485  void pop(GLuint index)
486  {
487  IndexedStack &stack = stackIndex_[index];
488  IndexedStackNode *top = stack.topNode();
489  IndexedStackNode *next = top->next_;
490  ValueType *indexedValue;
491  GLboolean valueChanged;
492 
493  // apply previously pushed indexed value
494  if(next) {
495  indexedValue = &next->value_.v;
496 
497  valueChanged = (top->value_.v != (*indexedValue));
498  stack.pop();
499  }
500  else { // last value. keep the node until next push
501  indexedValue = &top->value_.v;
502 
503  valueChanged = GL_FALSE;
504  stack.popKeepNode();
505  lastNodei_[index] = top;
506  doApplyi_[index] = &applyEmptyStampedi;
507  }
508  lastStampi_.pop();
509 
510  GLint lastStampi = (stack.isEmpty() ? -1 : stack.top().stamp);
511  // reset to equation with latest stamp
512  if(lastStamp_.top() > lastStampi) {
513  const ValueType &globalValue = stack_.top().v;
514  if((*indexedValue) != globalValue) applyi_(index,globalValue);
515  }
516  else if(valueChanged)
517  { applyi_(index, *indexedValue); }
518 
519  // count number of pushes for value stamps
520  counter_.x -= 1;
521  counter_.y -= 1;
522  }
