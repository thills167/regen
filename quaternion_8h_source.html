120  {

130  y = fCosRoll * fSinPitch * fCosYaw + fSinRoll * fCosPitch * fSinYaw;

132  w = fCosRoll * fCosPitchCosYaw + fSinRoll * fSinPitchSinYaw;
133  }

175  mat4x4.x[ 5] = -2.0f * (x * x + z * z) + 1.0f;
176  mat4x4.x[ 6] = 2.0f * (y * z - x * w);
177  mat4x4.x[ 8] = 2.0f * (x * z - y * w);
178  mat4x4.x[ 9] = 2.0f * (y * z + x * w);
179  mat4x4.x[10] = -2.0f * (x * x + y * y) + 1.0f;

190  inline void interpolate(
191  const Quaternion& pStart,
192  const Quaternion& pEnd,
193  GLfloat pFactor)
194  {
195  // calc cosine theta
196  GLfloat cosom =
197  pStart.x * pEnd.x +
198  pStart.y * pEnd.y +
199  pStart.z * pEnd.z +
200  pStart.w * pEnd.w;
201 
202  // adjust signs (if necessary)
203  Quaternion end = pEnd;
204  if(cosom < 0.0f)
205  {
206  cosom = -cosom;
207  end.x = -end.x; // Reverse all signs
208  end.y = -end.y;
209  end.z = -end.z;
210  end.w = -end.w;
211  }
212 
213  // Calculate coefficients
214  GLfloat sclp, sclq;
215  if( (1.0f - cosom) > 0.0001f) // 0.0001 -> some epsillon
216  {
217  // Standard case (slerp)
218  float omega, sinom;
219  omega = acos( cosom); // extract theta from dot product's cos theta
220  sinom = sin( omega);
221  sclp = sin( (1.0f - pFactor) * omega) / sinom;
222  sclq = sin( pFactor * omega) / sinom;
223  } else {
224  // Very close, do linear interp (because it's faster)
225  sclp = 1.0f - pFactor;
226  sclq = pFactor;
227  }
228 
229  x = sclp * pStart.x + sclq * end.x;
230  y = sclp * pStart.y + sclq * end.y;
231  z = sclp * pStart.z + sclq * end.z;
232  w = sclp * pStart.w + sclq * end.w;
233  }
