119  {
120  const GLfloat fSinPitch(sin(fPitch*0.5f));
121  const GLfloat fCosPitch(cos(fPitch*0.5f));
122  const GLfloat fSinYaw(sin(fYaw*0.5f));

124  const GLfloat fSinRoll(sin(fRoll*0.5f));

129  y = fCosRoll * fSinPitch * fCosYaw + fSinRoll * fCosPitch * fSinYaw;
130  z = fCosRoll * fCosPitch * fSinYaw - fSinRoll * fSinPitch * fCosYaw;
131  w = fCosRoll * fCosPitchCosYaw + fSinRoll * fSinPitchSinYaw;

168  {

189  inline void interpolate(
190  const Quaternion& pStart,
191  const Quaternion& pEnd,
192  GLfloat pFactor)
193  {
194  // calc cosine theta
195  GLfloat cosom =
196  pStart.x * pEnd.x +
197  pStart.y * pEnd.y +
198  pStart.z * pEnd.z +
199  pStart.w * pEnd.w;
200 
201  // adjust signs (if necessary)
202  Quaternion end = pEnd;
203  if(cosom < 0.0f)
204  {
205  cosom = -cosom;
206  end.x = -end.x; // Reverse all signs
207  end.y = -end.y;
208  end.z = -end.z;
209  end.w = -end.w;
210  }
211 
212  // Calculate coefficients
213  GLfloat sclp, sclq;
214  if( (1.0f - cosom) > 0.0001f) // 0.0001 -> some epsillon
215  {
216  // Standard case (slerp)
217  float omega, sinom;
218  omega = acos( cosom); // extract theta from dot product's cos theta
219  sinom = sin( omega);
220  sclp = sin( (1.0f - pFactor) * omega) / sinom;
221  sclq = sin( pFactor * omega) / sinom;
222  } else {
223  // Very close, do linear interp (because it's faster)
224  sclp = 1.0f - pFactor;
225  sclq = pFactor;
226  }
227 
228  x = sclp * pStart.x + sclq * end.x;
229  y = sclp * pStart.y + sclq * end.y;
230  z = sclp * pStart.z + sclq * end.z;
231  w = sclp * pStart.w + sclq * end.w;
232  }
